dummy variable

Dummy_I <- Season - 1

Q <- matrix(nrow =n, ncol = Dummy_I)
Dummy_Name <- vector("character", length = Dummy_I)

for(i in 1:Dummy_I){
  Dummy_Name[i] <- paste0("Q", i)
}

colnames(Q) <- Dummy_Name

for(i in 1:n){
  for(j in 1:Dummy_I){
    if(Quarter[i] == j){
      Q[i,j] <- 1
    } else {
      Q[i,j] <- 0
    }  
  }
}
Data.Subscribers <- cbind(t, Q)
